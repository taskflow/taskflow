185 // Operator ()
186 template <template <typename...> typename E>
187 void BasicTaskflow<E>::Closure::operator () () const {
188 
189  //assert(taskflow && node);
190 
191  // Here we need to fetch the num_successors first to avoid the invalid memory
192  // access caused by topology clear.
193  const auto num_successors = node->num_successors();
194 
195  // regular node type
196  // The default node work type. We only need to execute the callback if any.
197  if(auto index=node->_work.index(); index == 0) {
198  if(auto &f = std::get<StaticWork>(node->_work); f != nullptr){
199  std::invoke(f);
200  }
201  }
202  // subflow node type
203  // The first time we enter into the subflow context,
204  // "subnodes" must be empty.
205  // After executing the user's callback on subflow,
206  // there will be at least one node node used as "super source".
207  // The second time we enter this context there is no need
208  // to re-execute the work.
209  else {
210  //assert(std::holds_alternative<DynamicWork>(node->_work));
211 
212  if(!node->_subgraph.has_value()){
213  node->_subgraph.emplace(); // Initialize the _subgraph
214  }
215 
216  SubflowBuilder fb(*(node->_subgraph));
217 
218  bool empty_graph = node->_subgraph->empty();
219 
220  std::invoke(std::get<DynamicWork>(node->_work), fb);
221 
222  // Need to create a subflow
223  if(empty_graph) {
224 
225  auto& S = node->_subgraph->emplace_front([](){});
226 
227  S._topology = node->_topology;
228 
229  for(auto i = std::next(node->_subgraph->begin()); i != node->_subgraph->end(); ++i) {
230 
231  i->_topology = node->_topology;
232 
233  if(i->num_successors() == 0) {
234  i->precede(fb.detached() ? node->_topology->_target : *node);
235  }
236 
237  if(i->num_dependents() == 0) {
238  S.precede(*i);
239  }
240  }
241 
242  // this is for the case where subflow graph might be empty
243  if(!fb.detached()) {
244  S.precede(*node);
245  }
246 
247  taskflow->_schedule(S);
248 
249  if(!fb.detached()) {
250  return;
251  }
252  }
253  }
254 
255  // At this point, the node/node storage might be destructed.
256  for(size_t i=0; i<num_successors; ++i) {
257  if(--(node->_successors[i]->_dependents) == 0) {
258  taskflow->_schedule(*(node->_successors[i]));
259  }
260  }
261 }

416  std::vector<Closure> closures;
