480 // Procedure: _schedule
481 // The main procedure to schedule a give task node.
482 // Each task node has two types of tasks - regular and subflow.
483 inline void Executor::_schedule(Node* node) {
484 
485  // module node need another initialization
486  if(node->_module != nullptr && !node->is_spawned()) {
487  _init_module_node(node);
488  }
489 
490  //no worker thread available
491  if(_workers.size() == 0){
492  _invoke(0, node);
493  return;
494  }
495 
496  auto& pt = _per_thread();
497 
498  // caller is a worker to this pool
499  if(pt.pool == this) {
500  _workers[pt.worker_id].queue.push(node);
501  return;
502  }
503  // master threads
504  else {
505  //std::scoped_lock lock(_mutex);
506  _queue.push(node);
507  }
508 
509  _notifier.notify(false);
510 }

512 // Procedure: _schedule
513 // The main procedure to schedule a set of task nodes.
514 // Each task node has two types of tasks - regular and subflow.
515 inline void Executor::_schedule(PassiveVector<Node*>& nodes) {
516 
517  // We need to cacth the node count to avoid accessing the nodes
518  // vector while the parent topology is removed!
519  const auto num_nodes = nodes.size();
520 
521  if(num_nodes == 0) {
522  return;
523  }
524 
525  for(auto node : nodes) {
526  if(node->_module != nullptr && !node->is_spawned()) {
527  _init_module_node(node);
528  }
529  }
530 
531  //no worker thread available
532  if(_workers.size() == 0){
533  for(auto node: nodes){
534  _invoke(0, node);
535  }
536  return;
537  }
538 
539  auto& pt = _per_thread();
540 
541  if(pt.pool == this) {
542  for(size_t i=0; i<num_nodes; ++i) {
543  _workers[pt.worker_id].queue.push(nodes[i]);
544  }
545  return;
546  }
547 
548  {
549  //std::scoped_lock lock(_mutex);
550  for(size_t k=0; k<num_nodes; ++k) {
551  _queue.push(nodes[k]);
552  }
553  }
554 
555  size_t N = std::max(size_t{1}, std::min(_num_idlers.load(), num_nodes));
556 
557  if(N >= _workers.size()) {
558  _notifier.notify(true);
559  }
560  else {
561  for(size_t k=0; k<N; ++k) {
562  _notifier.notify(false);
563  }
564  }
565 }

567 // Procedure: _init_module_node
568 inline void Executor::_init_module_node(Node* node) {
569 
570  node->_work = [node=node, this, tgt{PassiveVector<Node*>()}] () mutable {
571 
572  // second time to enter this context
573  if(node->is_spawned()) {
574  node->_dependents.resize(node->_dependents.size()-tgt.size());
575  for(auto& t: tgt) {
576  t->_successors.clear();
577  }
578  return ;
579  }
580 
581  // first time to enter this context
582  node->set_spawned();
583 
584  PassiveVector<Node*> src;
585 
586  for(auto &n: node->_module->_graph) {
587  n._topology = node->_topology;
588  if(n.num_dependents() == 0) {
589  src.push_back(&n);
590  }
591  if(n.num_successors() == 0) {
592  n.precede(*node);
593  tgt.push_back(&n);
594  }
595  }
596 
597  _schedule(src);
598  };

601 // Procedure:
602 inline void Executor::_invoke(unsigned me, Node* node) {
603 
604  // Here we need to fetch the num_successors first to avoid the invalid memory
605  // access caused by topology clear.
606  const auto num_successors = node->num_successors();
607 
608  // regular node type
609  // The default node work type. We only need to execute the callback if any.
610  if(auto index=node->_work.index(); index == 0) {
611  if(node->_module != nullptr) {
612  bool first_time = !node->is_spawned();
613  _invoke_static_work(me, node);
614  if(first_time) {
615  return ;
616  }
617  }
618  else {
619  if(auto &f = std::get<Node::StaticWork>(node->_work); f != nullptr){
620  _invoke_static_work(me, node);
621  }
622  }
623  }
624  // subflow node type
625  else {
626 
627  // Clear the subgraph before the task execution
628  if(!node->is_spawned()) {
629  node->_subgraph.emplace();
630  }
631 
632  Subflow fb(*(node->_subgraph));
633 
634  _invoke_dynamic_work(me, node, fb);
635 
636  // Need to create a subflow if first time & subgraph is not empty
637  if(!node->is_spawned()) {
638  node->set_spawned();
639  if(!node->_subgraph->empty()) {
640  // For storing the source nodes
641  PassiveVector<Node*> src;
642  for(auto& n : *(node->_subgraph)) {
643  n._topology = node->_topology;
644  n.set_subtask();
645  if(n.num_successors() == 0) {
646  if(fb.detached()) {
647  node->_topology->_num_sinks ++;
648  }
649  else {
650  n.precede(*node);
651  }
652  }
653  if(n.num_dependents() == 0) {
654  src.push_back(&n);
655  }
656  }
657 
658  _schedule(src);
659 
660  if(!fb.detached()) {
661  return;
662  }
663  }
664  }
665  } // End of DynamicWork -----------------------------------------------------
666 
667  // Recover the runtime change due to dynamic tasking except the target & spawn tasks
668  // This must be done before scheduling the successors, otherwise this might cause
669  // race condition on the _dependents
670  //if(num_successors && !node->_subtask) {
671  if(!node->is_subtask()) {
672  // Only dynamic tasking needs to restore _dependents
673  // TODO:
674  if(node->_work.index() == 1 && !node->_subgraph->empty()) {
675  while(!node->_dependents.empty() && node->_dependents.back()->is_subtask()) {
676  node->_dependents.pop_back();
677  }
678  }
679  node->_num_dependents = static_cast<int>(node->_dependents.size());
680  node->unset_spawned();
681  }
682 
683  // At this point, the node storage might be destructed.
684  for(size_t i=0; i<num_successors; ++i) {
685  if(--(node->_successors[i]->_num_dependents) == 0) {
686  _schedule(node->_successors[i]);
687  }
688  }
689 
690  // A node without any successor should check the termination of topology
691  if(num_successors == 0) {
692  if(--(node->_topology->_num_sinks) == 0) {
693  if(_workers.size() > 0) { // finishing this topology
694  _tear_down_topology(*node->_topology);
695  }
696  }
697  }
698 }

753 // Function: _tear_down_topology
754 inline void Executor::_tear_down_topology(Topology& tpg) {
755 
756  auto &f = tpg._taskflow;
757 
758  // TODO: check whether we can use tpg instead of t._topologies.front()
759  // below and elsewhere.
760  //assert(&tpg == &(f._topologies.front()));
761 
762  // case 1: we still need to run the topology again
763  if(!std::invoke(tpg._pred)) {
764  tpg._recover_num_sinks();
765  _schedule(tpg._sources);
766  }
767  // case 2: the final run of this topology
768  else {
769 
770  if(tpg._call != nullptr) {
771  std::invoke(tpg._call);
772  }
773 
774  f._mtx.lock();
775 
776  // If there is another run (interleave between lock)
777  if(f._topologies.size() > 1) {
778 
779  // Set the promise
780  tpg._promise.set_value();
781  f._topologies.pop_front();
782  f._mtx.unlock();
783 
784  {
785  std::scoped_lock lock(_topology_mutex);
786  _num_topologies--;
787  }
788 
789  f._topologies.front()._bind(f._graph);
790  _schedule(f._topologies.front()._sources);
791  }
792  else {
793  assert(f._topologies.size() == 1);
794  // Need to back up the promise first here becuz taskflow might be
795  // destroy before taskflow leaves
796  auto p {std::move(tpg._promise)};
797 
798  f._topologies.pop_front();
799 
800  f._mtx.unlock();
801 
802  // We set the promise in the end in case taskflow leaves before taskflow
803  p.set_value();
804  {
805  std::scoped_lock lock(_topology_mutex);
806  _num_topologies--;
807  }
808  _topology_cv.notify_one();
809  }
810  }
811 }

813 // Function: run_until
814 template <typename P, typename C>
815 std::future<void> Executor::run_until(Taskflow& f, P&& pred, C&& c) {
816 
817  // Predicate must return a boolean value
818  static_assert(std::is_invocable_v<C> && std::is_invocable_v<P>);
819 
820  if(std::invoke(pred)) {
821  return std::async(std::launch::deferred, [](){});
822  }
823 
824  // Speicla case of zero workers needs
825  // - iterative execution to avoid stack overflow
826  // - avoid execution of last_work
827  if(_workers.size() == 0) {
828 
829  Topology tpg(f, std::forward<P>(pred), std::forward<C>(c));
830 
831  // Clear last execution data & Build precedence between nodes and target
832  tpg._bind(f._graph);
833 
834  do {
835  _schedule(tpg._sources);
836  tpg._recover_num_sinks();
837  } while(!std::invoke(tpg._pred));
838 
839  if(tpg._call != nullptr) {
840  std::invoke(tpg._call);
841  }
842 
843  return std::async(std::launch::deferred, [](){});
844  }
845 
846  // has worker(s)
847  {
848  std::lock_guard lock(_topology_mutex);
849  _num_topologies++;
850  }
851 
852  // Multi-threaded execution.
853  bool run_now {false};
854  Topology* tpg;
855  std::future<void> future;
856 
857  {
858  std::scoped_lock lock(f._mtx);
859 
860  // create a topology for this run
861  tpg = &(f._topologies.emplace_back(f, std::forward<P>(pred), std::forward<C>(c)));
