462 // Procedure: _exploit_task
463 inline void Executor::_exploit_task(Worker& w, Node*& t) {
464 
465  assert(!w.cache);
466 
467  if(t) {
468 
469  const auto d = w.domain;
470 
471  if(_num_actives[d].fetch_add(1) == 0 && _num_thieves[d] == 0) {
472  _notifier[d].notify(false);
473  }
474 
475  auto tpg = t->_topology;
476  auto par = t->_parent;
477  auto exe = size_t{1};
478 
479  do {
480  _invoke(w, t);
481 
482  if(w.cache) {
483  t = w.cache;
484  w.cache = nullptr;
485  }
486  else {
487  t = w.wsq[d].pop();
488  if(t) {
489  // We only increment the counter when poping task from wsq
490  // (NOT including cache!)
491  if(t->_parent == par) {
492  exe++;
493  }
494  // joined subflow
495  else {
496  if(par == nullptr) {
497  // still have tasks so the topology join counter can't be zero
498  t->_topology->_join_counter.fetch_sub(exe);
499  }
500  else {
501  auto ret = par->_join_counter.fetch_sub(exe);
502  if(ret == exe) {
503  if(par->domain() == d) {
504  w.wsq[d].push(par);
505  }
506  else {
507  _schedule(par, false);
508  }
509  }
510  }
511  exe = 1;
512  par = t->_parent;
513  }
514  }
515  else {
516  // If no more local tasks!
517  if(par == nullptr) {
518  if(tpg->_join_counter.fetch_sub(exe) == exe) {
519  // TODO: Store tpg in local variable not in w
520  _tear_down_topology(&tpg);
521  if(tpg != nullptr) {
522  t = w.wsq[d].pop();
523  if(t) {
524  exe = 1;
525  }
526  }
527  }
528  }
529  else {
530  if(par->_join_counter.fetch_sub(exe) == exe) {
531  if(par->domain() == d) {
532  t = par;
533  par = par->_parent;
534  exe = 1;
535  }
536  else {
537  _schedule(par, false);
538  }
539  }
540  }
541  }
542  }
543  } while(t);
544 
545  --_num_actives[d];
546  }

550 inline bool Executor::_wait_for_task(Worker& worker, Node*& t) {
551 
552  const auto d = worker.domain;
553 
554  wait_for_task:
555 
556  assert(!t);
557 
558  ++_num_thieves[d];
559 
560  explore_task:
561 
562  _explore_task(worker, t);
563 
564  if(t) {
565  if(_num_thieves[d].fetch_sub(1) == 1) {
566  _notifier[d].notify(false);
567  }
568  return true;
569  }
570 
571  _notifier[d].prepare_wait(worker.waiter);
572 
573  //if(auto vtm = _find_victim(me); vtm != _workers.size()) {
574  if(!_wsq[d].empty()) {
575 
576  _notifier[d].cancel_wait(worker.waiter);
577  //t = (vtm == me) ? _wsq.steal() : _workers[vtm].wsq.steal();
578 
579  t = _wsq[d].steal();
580  if(t) {
581  if(_num_thieves[d].fetch_sub(1) == 1) {
582  _notifier[d].notify(false);
583  }
584  return true;
585  }
586  else {
587  goto explore_task;
588  }
589  }
590 
591  if(_done) {
592  _notifier[d].cancel_wait(worker.waiter);
593  for(int i=0; i<NUM_DOMAINS; ++i) {
594  _notifier[i].notify(true);
595  }
596  --_num_thieves[d];
597  return false;
598  }
599 
600  if(_num_thieves[d].fetch_sub(1) == 1) {
601  if(_num_actives[d]) {
602  _notifier[d].cancel_wait(worker.waiter);
603  goto wait_for_task;
604  }
605  // check all domain queue again
606  for(auto& w : _workers) {
607  if(!w.wsq[d].empty()) {
608  _notifier[d].cancel_wait(worker.waiter);
609  goto wait_for_task;
610  }
611  }
612  }
613 
614  // Now I really need to relinguish my self to others
615  _notifier[d].commit_wait(worker.waiter);
616 
617  return true;
618 }

636 // The main procedure to schedule a give task node.
637 // Each task node has two types of tasks - regular and subflow.
638 inline void Executor::_schedule(Node* node, bool bypass_hint) {
639 
640  //assert(_workers.size() != 0);
641 
642  const auto d = node->domain();
643 
644  // caller is a worker to this pool
645  auto worker = _per_thread().worker;
646 
647  if(worker != nullptr && worker->executor == this) {
648  if(bypass_hint) {
649  assert(!worker->cache);
650  worker->cache = node;
651  }
652  else {
653  worker->wsq[d].push(node);
654  if(worker->domain != d) {
655  if(_num_actives[d] == 0 && _num_thieves[d] == 0) {
656  _notifier[d].notify(false);
657  }
658  }
659  }
660  return;
661  }
662 
663  // other threads
664  {
665  std::lock_guard<std::mutex> lock(_wsq_mutex);
666  _wsq[d].push(node);
667  }
668 
669  _notifier[d].notify(false);
670 }

672 // Procedure: _schedule
673 // The main procedure to schedule a set of task nodes.
674 // Each task node has two types of tasks - regular and subflow.
675 inline void Executor::_schedule(PassiveVector<Node*>& nodes) {
676 
677  //assert(_workers.size() != 0);
678 
679  // We need to cacth the node count to avoid accessing the nodes
680  // vector while the parent topology is removed!
681  const auto num_nodes = nodes.size();
682 
683  if(num_nodes == 0) {
684  return;
685  }
686 
687  // worker thread
688  auto worker = _per_thread().worker;
689 
690  // task counts
691  size_t tcount[NUM_DOMAINS] = {0};
692 
693  if(worker != nullptr && worker->executor == this) {
694  for(size_t i=0; i<num_nodes; ++i) {
695  const auto d = nodes[i]->domain();
696  worker->wsq[d].push(nodes[i]);
697  tcount[d]++;
698  }
699 
700  for(int d=0; d<NUM_DOMAINS; ++d) {
701  if(tcount[d] && d != worker->domain) {
702  if(_num_actives[d] == 0 && _num_thieves[d] == 0) {
703  _notifier[d].notify_n(tcount[d]);
704  }
705  }
706  }
707 
708  return;
709  }
710 
711  // other threads
712  {
713  std::lock_guard<std::mutex> lock(_wsq_mutex);
714  for(size_t k=0; k<num_nodes; ++k) {
715  const auto d = nodes[k]->domain();
716  _wsq[d].push(nodes[k]);
717  tcount[d]++;
718  }
719  }
720 
721  for(int d=0; d<NUM_DOMAINS; ++d) {
722  _notifier[d].notify_n(tcount[d]);
723  }
724 }

727 // Procedure: _invoke
728 inline void Executor::_invoke(Worker& worker, Node* node) {
729 
730  //assert(_workers.size() != 0);
731 
732  // Here we need to fetch the num_successors first to avoid the invalid memory
733  // access caused by topology clear.
734  const auto num_successors = node->num_successors();
735 
736  // acquire the parent flow counter
737  auto& c = (node->_parent) ? node->_parent->_join_counter :
738  node->_topology->_join_counter;
739 
740  // switch is faster than nested if-else due to jump table
741  switch(node->_handle.index()) {
742  // static task
743  case Node::STATIC_WORK:{
744  _invoke_static_work(worker, node);
745  }
746  break;
747  // module task
748  case Node::MODULE_WORK: {
749  bool first_time = !node->_has_state(Node::SPAWNED);
750  bool emptiness = false;
751  _set_up_module_work(node, emptiness);
752  if(first_time && !emptiness) {
753  return;
754  }
755  }
756  break;
757  // dynamic task
758  case Node::DYNAMIC_WORK: {
759 
760  // Need to create a subflow if it is the first time entering here
761  if(!node->_has_state(Node::SPAWNED)) {
762 
763  auto& subgraph = nstd::get<Node::DynamicWork>(node->_handle).subgraph;
764 
765  subgraph.clear();
766  Subflow fb(subgraph);
767 
768  _invoke_dynamic_work(worker, node, fb);
769 
770  node->_set_state(Node::SPAWNED);
771 
772  if(!subgraph.empty()) {
773 
774  PassiveVector<Node*> src;
775 
776  for(auto n : subgraph._nodes) {
777 
778  n->_topology = node->_topology;
779  n->_set_up_join_counter();
780 
781  if(!fb.detached()) {
782  n->_parent = node;
783  }
784 
785  if(n->num_dependents() == 0) {
786  src.push_back(n);
787  }
788  }
789 
790  const bool join = fb.joined();
791  if(!join) {
792  // Detach mode
793  node->_topology->_join_counter.fetch_add(src.size());
794  }
795  else {
796  // Join mode
797  node->_join_counter.fetch_add(src.size());
798 
799  // spawned node needs another second-round execution
800  c.fetch_add(1);
801  }
802 
803  _schedule(src);
804 
805  if(join) {
806  return;
807  }
808  } // End of first time
809  }
810  }
811  break;
812  // condition task
813  case Node::CONDITION_WORK: {
814 
815  if(node->_has_state(Node::BRANCH)) {
816  node->_join_counter = node->num_strong_dependents();
817  }
818  else {
819  node->_join_counter = node->num_dependents();
820  }
821 
822  int id;
823  _invoke_condition_work(worker, node, id);
824 
825  if(id >= 0 && static_cast<size_t>(id) < num_successors) {
826  auto s = node->_successors[id];
827  s->_join_counter.store(0);
828 
829  if(s->domain() == worker.domain) {
830  _schedule(s, true);
831  }
832  else {
833  c.fetch_add(1);
834  _schedule(s, false);
835  }
836  }
837  return ;
838  } // no need to add a break here due to the immediate return
839 
840  // cudaflow task
841 #ifdef TF_ENABLE_CUDA
842  case Node::CUDAFLOW_WORK: {
843  _invoke_cudaflow_work(worker, node);
844  }
845  break;
846 #endif
847 
848  // monostate
849  default:
850  break;
851  }
852 
853 
854  // We MUST recover the dependency since subflow may have
855  // a condition node to go back (cyclic).
856  // This must be done before scheduling the successors, otherwise this might cause
857  // race condition on the _dependents
858  if(node->_has_state(Node::BRANCH)) {
859  // If this is a case node, we need to deduct condition predecessors
860  node->_join_counter = node->num_strong_dependents();
861  }
862  else {
863  node->_join_counter = node->num_dependents();
864  }
865 
866  node->_unset_state(Node::SPAWNED);
867 
868  // At this point, the node storage might be destructed.
869  Node* cache {nullptr};
870 
871  for(size_t i=0; i<num_successors; ++i) {
872  if(--(node->_successors[i]->_join_counter) == 0) {
873  if(node->_successors[i]->domain() != worker.domain) {
874  c.fetch_add(1);
875  _schedule(node->_successors[i], false);
876  }
877  else {
878  if(cache) {
879  c.fetch_add(1);
880  _schedule(cache, false);
881  }
882  cache = node->_successors[i];
883  }
884  }
885  }
886 
887  if(cache) {
888  _schedule(cache, true);
889  }
890 }
