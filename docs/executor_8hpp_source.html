720 
721  // We need to cacth the node count to avoid accessing the nodes
722  // vector while the parent topology is removed!
723  const auto num_nodes = nodes.size();
724 
725  if(num_nodes == 0) {
726  return;
727  }
728 
729  // worker thread
730  auto worker = _per_thread().worker;
731 
732  // task counts
733  size_t tcount[NUM_DOMAINS] = {0};
734 
735  if(worker != nullptr && worker->executor == this) {
736  for(size_t i=0; i<num_nodes; ++i) {
737  const auto d = nodes[i]->domain();
738  worker->wsq[d].push(nodes[i]);
739  tcount[d]++;

741 
742  for(int d=0; d<NUM_DOMAINS; ++d) {
743  if(tcount[d] && d != worker->domain) {
744  if(_num_actives[d] == 0 && _num_thieves[d] == 0) {
745  _notifier[d].notify_n(tcount[d]);
746  }
747  }
748  }
749 
750  return;
751  }
752 
753  // other threads
754  {
755  std::lock_guard<std::mutex> lock(_wsq_mutex);
756  for(size_t k=0; k<num_nodes; ++k) {
757  const auto d = nodes[k]->domain();
758  _wsq[d].push(nodes[k]);
759  tcount[d]++;
760  }
761  }
762 
763  for(int d=0; d<NUM_DOMAINS; ++d) {
764  _notifier[d].notify_n(tcount[d]);
765  }
766 }

769 // Procedure: _invoke
770 inline void Executor::_invoke(Worker& worker, Node* node) {
771 
772  //assert(_workers.size() != 0);
773 
774  // Here we need to fetch the num_successors first to avoid the invalid memory
775  // access caused by topology clear.
776  const auto num_successors = node->num_successors();
777 
778  // task type
779  auto type = node->_handle.index();
780 
781  // condition task
782  int cond = -1;
783 
784  // switch is faster than nested if-else due to jump table
785  switch(type) {
786  // static task
787  case Node::STATIC_TASK:{
788  _invoke_static_task(worker, node);
789  }
790  break;
791 
792  // dynamic task
793  case Node::DYNAMIC_TASK: {
794  _invoke_dynamic_task(worker, node);
795  }
796  break;
797 
798  // condition task
799  case Node::CONDITION_TASK: {
800  _invoke_condition_task(worker, node, cond);
801  }
802  break;
803  //} // no need to add a break here due to the immediate return
804 
805  // module task
806  case Node::MODULE_TASK: {
807  _invoke_module_task(worker, node);
808  }
809  break;
810 
811  // async task
812  case Node::ASYNC_TASK: {
813  _invoke_async_task(worker, node);
814  _decrement_topology_and_notify();
815  return ;
816  }
817  break;
818 
819  // cudaflow task
820 #ifdef TF_ENABLE_CUDA
821  case Node::CUDAFLOW_TASK: {
822  _invoke_cudaflow_task(worker, node);
823  }
824  break;
825 #endif
826 
827  // monostate
828  default:
829  break;
830  }
831 
832  // We MUST recover the dependency since the graph may have cycles.
833  // This must be done before scheduling the successors, otherwise this might cause
834  // race condition on the _dependents
835  if(node->_has_state(Node::BRANCHED)) {
836  node->_join_counter = node->num_strong_dependents();
837  }
838  else {
839  node->_join_counter = node->num_dependents();
840  }
841 
842  // acquire the parent flow counter
843  auto& c = (node->_parent) ? node->_parent->_join_counter :
844  node->_topology->_join_counter;
845 
846  // At this point, the node storage might be destructed (to be verified)
847  // case 1: non-condition task
848  if(type != Node::CONDITION_TASK) {
849  for(size_t i=0; i<num_successors; ++i) {
850  if(--(node->_successors[i]->_join_counter) == 0) {
851  c.fetch_add(1);
852  _schedule(node->_successors[i]);
853  }
854  }
855  }
856  // case 2: condition task
857  else {
858  if(cond >= 0 && static_cast<size_t>(cond) < num_successors) {
859  auto s = node->_successors[cond];
860  s->_join_counter.store(0); // seems redundant but just for invariant
861  c.fetch_add(1);
862  _schedule(s);
863  }
864  }
865 
866  // tear down topology if the node is the last one
867  if(node->_parent == nullptr) {
868  if(node->_topology->_join_counter.fetch_sub(1) == 1) {
869  _tear_down_topology(node->_topology);
870  }
871  }
872  else { // joined subflow
873  node->_parent->_join_counter.fetch_sub(1);
874  }
875 }
