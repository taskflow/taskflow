160 
161  assert(is_owner());

165 
166  _exiting = true;
167 
168  for(auto w : _idlers){
169  w->ready = true;
170  w->task = std::nullopt;
171  w->cv.notify_one();
172  }
173  _idlers.clear();
174  }

176  for(auto& t : _threads){
177  t.join();
178  }

187 // Function: spawn
188 template <typename Closure>
189 void SpeculativeThreadpool<Closure>::_spawn(unsigned N) {
190 
191  assert(is_owner() && _workers.size() == N);
192 
193  // Lock to synchronize all workers before creating _worker_mapss
194  std::scoped_lock lock(_mutex);
195 
196  for(size_t i=0; i<N; ++i){
197 
198  _threads.emplace_back([this, &w=_workers[i]]() -> void {
199 
200  std::optional<Closure> t;
201 
202  std::unique_lock lock(_mutex);
203 
204  while(!_exiting){
205  if(_tasks.empty()){
206  w.ready = false;
207  _idlers.push_back(&w);
208 
209  while(!w.ready) {
210  w.cv.wait(lock);
211  }
212 
213  t = std::move(w.task);
214  w.task = std::nullopt;
215  }
216  else{
217  t = std::move(_tasks.back());
218  _tasks.pop_back();
219  }
220 
221  if(t) {
222  lock.unlock();
223  // speculation loop
224  while(t) {
225  (*t)();
226  if(w.task) {
227  t = std::move(w.task);
228  w.task = std::nullopt;
229  }
230  else {
231  t = std::nullopt;
232  }
233  }
234  lock.lock();
235  }
236  }
237  });
238 
239  _worker_maps[_threads[i].get_id()] = &_workers[i];
240  } // End of For ---------------------------------------------------------------------------------
241 }

243 template <typename Closure>
244 template <typename... ArgsT>
245 void SpeculativeThreadpool<Closure>::emplace(ArgsT&&... args) {
246 
247  //no worker thread available
248  if(num_workers() == 0){
249  Closure{std::forward<ArgsT>(args)...}();
250  return;
251  }
252 
253  // speculation
254  auto tid = std::this_thread::get_id();
255 
256  if(tid != _owner){
257  auto iter = _worker_maps.find(tid);
258  if(iter != _worker_maps.end() && !(iter->second->task)){
259  iter->second->task.emplace(std::forward<ArgsT>(args)...);
260  return ;
261  }
262  }
263 
264  std::scoped_lock lock(_mutex);
265  if(_idlers.empty()){
266  _tasks.emplace_back(std::forward<ArgsT>(args)...);
267  }
268  else{
269  Worker* w = _idlers.back();
270  _idlers.pop_back();
271  w->ready = true;
272  w->task.emplace(std::forward<ArgsT>(args)...);
273  w->cv.notify_one();
274  }
275 }

278 template <typename Closure>
279 void SpeculativeThreadpool<Closure>::batch(std::vector<Closure>& tasks){
280 
281  if(tasks.empty()) {
282  return;
283  }
284 
285  //no worker thread available
286  if(num_workers() == 0){
287  for(auto& c: tasks){
288  c();
289  }
290  return;
291  }
292 
293  size_t consumed {0};
294 
295  // speculation
296  if(std::this_thread::get_id() != _owner){
297  auto iter = _this_worker();
298  if(iter != _worker_maps.end() && !(iter->second->task)){
299  iter->second->task.emplace(std::move(tasks[consumed++]));
300  if(tasks.size() == consumed) {
301  return ;
302  }
303  }
304  }
305 
306  std::scoped_lock lock(_mutex);
307  while(!_idlers.empty() && tasks.size() != consumed) {
308  Worker* w = _idlers.back();
309  _idlers.pop_back();
310  w->ready = true;
311  w->task.emplace(std::move(tasks[consumed ++]));
312  w->cv.notify_one();
313  }
314 
315  if(tasks.size() == consumed) return ;
316  _tasks.reserve(_tasks.size() + tasks.size() - consumed);
317  std::move(tasks.begin()+consumed, tasks.end(), std::back_inserter(_tasks));
318 }
