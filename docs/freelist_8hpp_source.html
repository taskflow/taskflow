134 void push(T node) {
135
136 TaggedPointer curr = _head.load(std::memory_order_relaxed);
137
138 while(true) {
139 _link(node) = curr.ptr;
140 TaggedPointer next = {node, curr.tag + 1};
141
142 if(_head.compare_exchange_weak(curr, next,
143 std::memory_order_release,
144 std::memory_order_relaxed)) {
145 break;
146 }
147 // on failure curr is reloaded by compare_exchange_weak
148 }

178 template <typename I>
179 void bulk_push(I first, size_t N) {
180
181 if(N == 0) {
182 return;
183 }
184
185 // chain nodes locally through their link fields (no atomics needed)
186 for(size_t i = 0; i < N-1; ++i) {
187 _link(first[i]) = first[i+1];
188 }
189 _link(first[N-1]) = nullptr;
190
191 TaggedPointer curr = _head.load(std::memory_order_relaxed);
192
193 while(true) {
194 // splice: tail of new chain points to current head
195 _link(first[N-1]) = curr.ptr;
196 TaggedPointer next = {first[0], curr.tag + 1};
197
198 if(_head.compare_exchange_weak(curr, next,
199 std::memory_order_release,
200 std::memory_order_relaxed)) {
201 break;
202 }
203 }
204 }
205
235 T pop() {
236
237 TaggedPointer curr = _head.load(std::memory_order_acquire);
238
239 while(curr.ptr != nullptr) {
240 TaggedPointer next = {_link(curr.ptr), curr.tag + 1};
241
242 if(_head.compare_exchange_weak(curr, next,
243 std::memory_order_release,
244 std::memory_order_acquire)) {
245 _link(curr.ptr) = nullptr; // clear link before returning to caller
246 return curr.ptr;
247 }
248 // on failure curr is reloaded
249 }
250 return nullptr;
251 }
