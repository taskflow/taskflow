357 }

442 template <typename F>
443 void loop(
444 size_t N, size_t W, std::atomic<size_t>& next, F&& func
445 ) const {
446
447 size_t chunk_size = (this->_chunk_size == 0) ? size_t{1} : this->_chunk_size;
448
449 size_t p1 = 2 * W * (chunk_size + 1);
450 float p2 = 0.5f / static_cast<float>(W);
451 size_t curr_b = next.load(std::memory_order_relaxed);
452
453 while(curr_b < N) {
454
455 size_t r = N - curr_b;
456
457 // fine-grained
458 if(r < p1) {
459 while(1) {
460 curr_b = next.fetch_add(chunk_size, std::memory_order_relaxed);
461 if(curr_b >= N) {
462 return;
463 }
464 func(curr_b, (std::min)(curr_b + chunk_size, N));
465 }
466 break;
467 }
468 // coarse-grained
469 else {
470 size_t q = static_cast<size_t>(p2 * r);
471 if(q < chunk_size) {
472 q = chunk_size;
473 }
474 //size_t curr_e = (q <= r) ? curr_b + q : N;
475 size_t curr_e = (std::min)(curr_b + q, N);
476 if(next.compare_exchange_strong(curr_b, curr_e, std::memory_order_relaxed,
477 std::memory_order_relaxed)) {
478 func(curr_b, curr_e);
479 //curr_b = next.load(std::memory_order_relaxed);
480 curr_b = curr_e;
481 }
482 }
483 }

540 template <IndexRangeMDLike R, typename F>
541 void loop(R& range, size_t N, size_t W, std::atomic<size_t>& next, F&& func) const {
542
543 size_t chunk_size = (this->_chunk_size == 0) ? size_t{1} : this->_chunk_size;
544 size_t requested_chunk_size;
545 size_t p1 = 2 * W * (chunk_size + 1);
546 float p2 = 0.5f / static_cast<float>(W);
547 size_t curr_b = next.load(std::memory_order_relaxed);
548
549 while(curr_b < N) {
550
551 size_t r = N - curr_b;
552
553 // find-grained
554 if(r < p1) {
555 requested_chunk_size = chunk_size;
556 }
557 // coarse
558 else {
559 requested_chunk_size = (std::max)(static_cast<size_t>(p2*r), chunk_size);
560 }
561
562 auto [box, consumed] = range.slice_ceil(curr_b, requested_chunk_size);
563
564 if (next.compare_exchange_weak(curr_b, curr_b + consumed,
565 std::memory_order_relaxed,
566 std::memory_order_relaxed)) {
567 func(box);
568 curr_b += consumed;
569 }
570 }
571 }

807 std::pair<size_t, size_t> chunk_size_range(size_t N, size_t W) const {
808
809 size_t b1 = static_cast<size_t>(_alpha * N * W);
810 size_t b2 = static_cast<size_t>(_beta * N * W);
811
812 if(b1 > b2) {
813 std::swap(b1, b2);
814 }
815
816 b1 = (std::max)(b1, size_t{1});
817 b2 = (std::max)(b2, b1 + 1);
818
819 return {b1, b2};
820 }

829 template <typename F>
830 void loop(
831 size_t N, size_t W, std::atomic<size_t>& next, F&& func
832 ) const {
833
834 auto [b1, b2] = chunk_size_range(N, W);
835
836 std::default_random_engine engine {std::random_device{}()};
837 std::uniform_int_distribution<size_t> dist(b1, b2);
838
839 size_t chunk_size = dist(engine);
840 size_t curr_b = next.fetch_add(chunk_size, std::memory_order_relaxed);
841
842 while(curr_b < N) {
843 func(curr_b, (std::min)(curr_b + chunk_size, N));
844 chunk_size = dist(engine);
845 curr_b = next.fetch_add(chunk_size, std::memory_order_relaxed);

893 curr_b += consumed;
