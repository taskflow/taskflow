27 // Whenever a job is inserted into the executor, the executor will check if there
28 // are any spare threads available. The spare thread will be woken through its local
29 // condition variable. The new job will be directly moved into
30 // this thread instead of pushed at the back of the pending queue.

174 void ProactiveExecutor<Closure>::_shutdown() {
175 
176  assert(is_owner());
177 
178  {
179  std::unique_lock lock(_mutex);
180 
181  _exiting = true;
182 
183  // we need to clear the workers under lock
184  for(auto w : _idlers){
185  w->ready = true;
186  w->task = std::nullopt;
187  w->cv.notify_one();
188  }
189  _idlers.clear();
190  }
191 
192  for(auto& t : _threads){
193  t.join();
194  }
195  _threads.clear();
196 
197  _exiting = false;

200 // Procedure: spawn
201 template <typename Closure>
202 void ProactiveExecutor<Closure>::_spawn(unsigned N) {
203 
204  assert(is_owner());
205 
206  for(size_t i=0; i<N; ++i){
207 
208  _threads.emplace_back([this, me=i] () -> void {
209 
210  Worker w;
211 
212  std::unique_lock lock(_mutex);
213 
214  while(!_exiting) {
215 
216  if(_tasks.empty()){
217 
218  w.ready = false;
219  _idlers.push_back(&w);
220 
221  while(!w.ready) {
222  w.cv.wait(lock);
223  }
224 
225  // shutdown cannot have task
226  if(w.task) {
227  lock.unlock();
228 
229  if(_observer) {
230  _observer->on_entry(me);
231  }
232 
233  (*w.task)();
234  w.task = std::nullopt;
235 
236  if(_observer) {
237  _observer->on_exit(me);
238  }
239 
240  lock.lock();
241  }
242  }
243  else{
244  Closure t{std::move(_tasks.back())};
245  _tasks.pop_back();
246  lock.unlock();
247 
248  if(_observer) {
249  _observer->on_entry(me);
250  }
251 
252  t();
253 
254  if(_observer) {
255  _observer->on_exit(me);
256  }
257 
258  lock.lock();
259  }
260  }
261  });
262 
263  }
264 }

269 void ProactiveExecutor<Closure>::emplace(ArgsT&&... args) {
270 
271  //no worker thread available
272  if(num_workers() == 0){
273  Closure{std::forward<ArgsT>(args)...}();
274  }
275  // ask one worker to run the task
276  else {
277  std::scoped_lock lock(_mutex);
278  if(_idlers.empty()){

283  _idlers.pop_back();

292 // Procedure: batch
293 template <typename Closure>
294 void ProactiveExecutor<Closure>::batch(std::vector<Closure>& tasks) {
295 
296  //no worker thread available
297  if(num_workers() == 0){
298  for(auto& t: tasks){
299  t();
300  }
301  }
302  // ask one worker to run the task
303  else {
304  size_t consumed {0};
305  std::scoped_lock lock(_mutex);
306  if(_idlers.empty()){
307  std::move(tasks.begin(), tasks.end(), std::back_inserter(_tasks));
308  return ;
309  }
310  else{
311  while(consumed != tasks.size() && !_idlers.empty()) {
312  Worker* w = _idlers.back();
313  _idlers.pop_back();
314  w->ready = true;
315  w->task.emplace(std::move(tasks[consumed++]));
316  w->cv.notify_one();
317  }
318  }
319  if(consumed == tasks.size()) return ;
320  _tasks.reserve(_tasks.size() + tasks.size() - consumed);
321  std::move(tasks.begin()+consumed, tasks.end(), std::back_inserter(_tasks));
322  }
323 }
