294 // Function: run_until
295 template <template <typename...> typename E>
296 template <typename P, typename C>
297 std::shared_future<void> BasicTaskflow<E>::run_until(Framework& f, P&& predicate, C&& c) {
298 
299  // Predicate must return a boolean value
300  static_assert(std::is_invocable_v<C> && std::is_invocable_v<P>);
301 
302  if(std::invoke(predicate)) {
303  return std::async(std::launch::deferred, [](){}).share();
304  }
305 
306  // create a topology for this run
307  auto &tpg = _topologies.emplace_back(f, std::forward<P>(predicate));
308 
309  // Iterative execution to avoid stack overflow
310  if(num_workers() == 0) {
311 
312  // Clear last execution data & Build precedence between nodes and target
313  tpg._bind(f._graph);
314 
315  do {
316  _schedule(tpg._sources);
317  tpg._recover_num_sinks();
318  } while(!std::invoke(tpg._predicate));
319 
320  std::invoke(c);
321  tpg._promise.set_value();
322 
323  return tpg._future;
324  }
325 
326  // Multi-threaded execution.
327  std::scoped_lock lock(f._mtx);
328 
329  f._topologies.push_back(&tpg);
330 
331  bool run_now = (f._topologies.size() == 1);
332 
333  if(run_now) {
334  tpg._bind(f._graph);
335  }
336 
337  tpg._work = [&f, c=std::forward<C>(c), this] () mutable {
338 
339  // case 1: we still need to run the topology again
340  if(!std::invoke(f._topologies.front()->_predicate)) {
341  f._topologies.front()->_recover_num_sinks();
342  _schedule(f._topologies.front()->_sources);
343  }
344  // case 2: the final run of this topology
345  else {
346  std::invoke(c);
347 
348  f._mtx.lock();
349 
350  // If there is another run (interleave between lock)
351  if(f._topologies.size() > 1) {
352 
353  // Set the promise
354  f._topologies.front()->_promise.set_value();
355  f._topologies.pop_front();
356  f._topologies.front()->_bind(f._graph);
357  f._mtx.unlock();
358  _schedule(f._topologies.front()->_sources);
359  }
360  else {
361  assert(f._topologies.size() == 1);
362  // Need to back up the promise first here becuz framework might be
363  // destroy before taskflow leaves
364  auto &p = f._topologies.front()->_promise;
365  f._topologies.pop_front();
366  f._mtx.unlock();
367 
368  // We set the promise in the end in case framework leaves before taskflow
369  p.set_value();
370  }
371  }
372  };
373 
374  if(run_now) {
375  _schedule(tpg._sources);
376  }
377 
378  return tpg._future;
379 }

387 // Operator ()
388 template <template <typename...> typename E>
389 void BasicTaskflow<E>::Closure::operator () () const {
390 
391  // Here we need to fetch the num_successors first to avoid the invalid memory
392  // access caused by topology clear.
393  const auto num_successors = node->num_successors();
394 
395  // regular node type
396  // The default node work type. We only need to execute the callback if any.
397  if(auto index=node->_work.index(); index == 0) {
398  if(node->_module != nullptr) {
399  bool first_time = !node->is_spawned();
400  std::invoke(std::get<StaticWork>(node->_work));
401  if(first_time) {
402  return ;
403  }
404  }
405  else {
406  if(auto &f = std::get<StaticWork>(node->_work); f != nullptr){
407  std::invoke(f);
408  }
409  }
410  }
411  // subflow node type
412  else {
413 
414  // Clear the subgraph before the task execution
415  if(!node->is_spawned()) {
416  node->_subgraph.emplace();
417  }
418 
419  SubflowBuilder fb(*(node->_subgraph));
420 
421  std::invoke(std::get<DynamicWork>(node->_work), fb);
422 
423  // Need to create a subflow if first time & subgraph is not empty
424  if(!node->is_spawned()) {
425  node->set_spawned();
426  if(!node->_subgraph->empty()) {
427  // For storing the source nodes
428  PassiveVector<Node*> src;
429  for(auto& n : *(node->_subgraph)) {
430  n._topology = node->_topology;
431  n.set_subtask();
432  if(n.num_successors() == 0) {
433  if(fb.detached()) {
434  node->_topology->_num_sinks ++;
435  }
436  else {
437  n.precede(*node);
438  }
439  }
440  if(n.num_dependents() == 0) {
441  src.push_back(&n);
442  }
443  }
444 
445  taskflow->_schedule(src);
446 
447  if(!fb.detached()) {
448  return;
449  }
450  }
451  }
452  } // End of DynamicWork -----------------------------------------------------
453 
454  // Recover the runtime change due to dynamic tasking except the target & spawn tasks
455  // This must be done before scheduling the successors, otherwise this might cause
456  // race condition on the _dependents
457  //if(num_successors && !node->_subtask) {
458  if(!node->is_subtask()) {
459  // Only dynamic tasking needs to restore _dependents
460  // TODO:
461  if(node->_work.index() == 1 && !node->_subgraph->empty()) {
462  while(!node->_dependents.empty() && node->_dependents.back()->is_subtask()) {
463  node->_dependents.pop_back();
464  }
465  }
466  node->_num_dependents = node->_dependents.size();
467  node->unset_spawned();
468  }
469 
470  // At this point, the node storage might be destructed.
471  for(size_t i=0; i<num_successors; ++i) {
472  if(--(node->_successors[i]->_num_dependents) == 0) {
473  taskflow->_schedule(*(node->_successors[i]));
474  }
475  }
476 
477  // A node without any successor should check the termination of topology
478  if(num_successors == 0) {
479  if(--(node->_topology->_num_sinks) == 0) {
480 
481  // This is the last executing node
482  bool is_framework = node->_topology->_handle.index() == 1;
483  if(node->_topology->_work != nullptr) {
484  std::invoke(node->_topology->_work);
485  }
486  if(!is_framework) {
487  node->_topology->_promise.set_value();
488  }
489  }
490  }
491 }

644 // Procedure: _schedule
645 // The main procedure to schedule a set of task nodes.
646 // Each task node has two types of tasks - regular and subflow.
647 template <template <typename...> typename E>
648 void BasicTaskflow<E>::_schedule(PassiveVector<Node*>& nodes) {
649  std::vector<Closure> closures;
650  closures.reserve(nodes.size());
651  for(auto src : nodes) {
652  // TODO: remove the assert?
653  if(src->_module != nullptr && !src->is_spawned()) {
654  _set_module_node(*src);
655  }
656  closures.emplace_back(*this, *src);
657  }
658  _executor->batch(closures);
659 }

662 template <template <typename...> typename E>
663 void BasicTaskflow<E>::_set_module_node(Node& node) {
664 
665  node._work = [&node, this, tgt{PassiveVector<Node*>()}] () mutable {
666 
667  // second time to enter this context
668  if(node.is_spawned()) {
669  node._dependents.resize(node._dependents.size()-tgt.size());
670  for(auto& t: tgt) {
671  t->_successors.clear();
672  }
673  return ;
674  }
675 
676  // first time to enter this context
677  node.set_spawned();
678 
679  PassiveVector<Node*> src;
680 
681  for(auto &n: node._module->_graph) {
682  n._topology = node._topology;
683  if(n.num_dependents() == 0) {
684  src.push_back(&n);
685  }
686  if(n.num_successors() == 0) {
687  n.precede(node);
688  tgt.push_back(&n);
689  }
690  }
691 
692  _schedule(src);
693  };
694 }
