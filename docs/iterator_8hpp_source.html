70template <std::integral T>
71constexpr size_t distance(T beg, T end, T step) {
72 if constexpr (std::is_unsigned_v<T>) {
73 // step is always positive for unsigned types — standard ceiling division
74 return static_cast<size_t>((end - beg + step - T{1}) / step);
75 } else {
76 // signed: step may be positive or negative
77 return static_cast<size_t>((end - beg + step + (step > T{0} ? T{-1} : T{1})) / step);
78 }
79}

97// Primary template: N-dimensional index range (N > 1)
98//
99// Represents the Cartesian product of N 1D IndexRange<T, 1> objects.
100//
101// Iteration order is row-major (last dimension is innermost / fastest),
102// matching the natural loop nesting:
103//
104// for i in dim[0]: // outermost

106// ...

249 std::array<T, N> coords(size_t flat) const {
250 std::array<T, N> c;
251 for (size_t d = N; d-- > 0; ) {
252 size_t sz = _dims[d].size();
253 size_t pos = flat % sz;
254 c[d] = _dims[d].begin() + static_cast<T>(pos) * _dims[d].step_size();
255 flat /= sz;
256 }

309 std::pair<IndexRange<T, N>, size_t> consume_chunk(size_t flat_beg, size_t requested_size) const {
310
311 if (requested_size == 0) {
312 return { *this, 0 };
313 }
314
315 std::array<size_t, N> coords;
316 size_t temp = flat_beg;
317
318 // 1. Convert flat_beg to ND coordinates
319 // We use the standard unsigned reverse iteration idiom (d-- > 0) to avoid
320 // -Wsign-compare warnings while correctly wrapping around after 0.
321 for (size_t d = N; d-- > 0; ) {
322 size_t dim_size = _dims[d].size();
323 coords[d] = temp % dim_size;
324 temp /= dim_size;
325 }
326
327 // 2. Find the highest valid dimension to grow (The "Trailing Zeros" Rule)
328 size_t grow_dim = N - 1;
329 size_t inner_volume = 1;
330 size_t active_inner_vol = 1; // Track the volume strictly inner to grow_dim
331
332 for (size_t d = N; d-- > 0; ) {
333 // The "Trailing Zeros" boundary check:
334 // We mathematically CANNOT grow dimension `d` if any dimension inner to it
335 // (d+1) is not sitting at coordinate 0. If we did, we would break the
336 // orthogonal hyper-rectangle shape and create a discontiguous "stair-step" box.
337 if (d + 1 < N && coords[d + 1] != 0) {
338 break;
339 }
340
341 // We are authorized to grow this dimension if one whole step (inner_volume)
342 // does not overshoot the requested chunk budget.
343 if (inner_volume <= requested_size) {
344 grow_dim = d;
345 active_inner_vol = inner_volume;
346 }
347 else {
348 // We hit the budget ceiling.
349 // Because inner_volume strictly increases as we move outward,
350 // all outer dimensions will also exceed the budget. We can safely stop looking.
351 break;
352 }
353
354 // Branchless accumulation: We multiply unconditionally. The CPU ALU is faster
355 // than checking `if (d > 0)`. The dead-store/potential-overflow when d == 0
356 // is completely harmless because the loop ends immediately after.
357 inner_volume *= _dims[d].size();
358 }
359
360 // 3. Determine how many steps to take along grow_dim
361 // We use integer floor division (/) here intentionally to decide steps needed.
362 // A step along `grow_dim` is an indivisible unit of work (active_inner_vol).
363 // Using ceiling division would routinely cause us to overshoot the requested_size
364 // budget. Floor division ensures we take as many *whole* slices as possible.
365 // The std::max(1, ...) handles the edge case where the budget is smaller than a
366 // single slice; we MUST take at least 1 step to guarantee forward progress.
367 size_t steps_left = _dims[grow_dim].size() - coords[grow_dim];
368 size_t steps_needed = (std::max)(size_t{1}, requested_size / active_inner_vol);
369 size_t steps_to_take = (std::min)(steps_left, steps_needed);
370
371 // 4. Construct the beautifully orthogonal Box
372 std::array<IndexRange<T, 1>, N> box_dims;
373 for(size_t d = 0; d < N; ++d) {
374 if (d < grow_dim) {
375 // Outer dimensions are locked at their current coordinate.
376 // We use `coords[d] + 1` for the end bound because C++ uses half-open
377 // intervals [begin, end). To lock a dimension to exactly ONE element,
378 // the end bound must be exactly 1 step past the begin bound.
379 box_dims[d] = IndexRange<T, 1>(
380 _dims[d].begin() + static_cast<T>(coords[d]) * _dims[d].step_size(),
381 _dims[d].begin() + static_cast<T>(coords[d] + 1) * _dims[d].step_size(),
382 _dims[d].step_size()
383 );
384 } else if (d == grow_dim) {
385 // The chosen dimension expands by the calculated steps.
386 box_dims[d] = IndexRange<T, 1>(
387 _dims[d].begin() + static_cast<T>(coords[d]) * _dims[d].step_size(),
388 _dims[d].begin() + static_cast<T>(coords[d] + steps_to_take) * _dims[d].step_size(),
389 _dims[d].step_size()
390 );
391 } else {
392 // Inner dimensions take their full unadulterated extent.
393 box_dims[d] = _dims[d];
394 }
395 }
396
397 return { IndexRange<T, N>(box_dims), steps_to_take * active_inner_vol };
398 }
399
400 private:
401
402 std::array<IndexRange<T, 1>, N> _dims;
403};
