70template <std::integral T>
71constexpr size_t distance(T beg, T end, T step) {
72 if constexpr (std::is_unsigned_v<T>) {
73 // step is always positive for unsigned types — standard ceiling division
74 return static_cast<size_t>((end - beg + step - T{1}) / step);
75 } else {
76 // signed: step may be positive or negative
77 return static_cast<size_t>((end - beg + step + (step > T{0} ? T{-1} : T{1})) / step);
78 }

508template <std::integral T, size_t N>
509size_t IndexRange<T, N>::ceil(size_t chunk_size) const {
510 // Pass 1: cache all dim sizes (one .size() call each) and find d_active.
511 size_t dim_sizes[N];
512 size_t d_active = N;
513 for (size_t d = 0; d < N; ++d) {
514 dim_sizes[d] = _dims[d].size();
515 if (dim_sizes[d] == 0) { d_active = d; break; }
516 }
517 if (d_active == 0) return 0;
518
519 // Pass 2: walk suffix products backward within [0, d_active) only.
520 // Return the first suffix product >= chunk_size, capped at size().
521 size_t inner_volume = 1;
522 if (inner_volume >= chunk_size) return inner_volume;
523 for (size_t d = d_active; d-- > 0; ) {
524 inner_volume *= dim_sizes[d];
525 if (inner_volume >= chunk_size) return inner_volume;
526 }
527 return inner_volume; // == size() of active dims, capped
528}

530template <std::integral T, size_t N>
531size_t IndexRange<T, N>::floor(size_t chunk_size) const {
532 // Pass 1: cache all dim sizes (one .size() call each) and find d_active.
533 size_t dim_sizes[N];
534 size_t d_active = N;
535 for (size_t d = 0; d < N; ++d) {
536 dim_sizes[d] = _dims[d].size();
537 if (dim_sizes[d] == 0) { d_active = d; break; }
538 }
539 if (d_active == 0) return 0;
540
541 // Pass 2: walk suffix products backward within [0, d_active) only.
542 // Return the largest suffix product <= chunk_size.
543 size_t inner_volume = 1;
544 size_t last_fit = 1;
545 for (size_t d = d_active; d-- > 0; ) {
546 size_t next = inner_volume * dim_sizes[d];
547 if (next > chunk_size) break;
548 inner_volume = next;
549 last_fit = inner_volume;
550 }
551 return last_fit;
552}

554template <std::integral T, size_t N>
555std::pair<IndexRange<T, N>, size_t>
556IndexRange<T, N>::slice_ceil(size_t flat_beg, size_t chunk_size) const {
557
558 if (chunk_size == 0) {
559 return { *this, 0 };
560 }
561
562 // Find d_active: the index of the first zero-size dimension (or N if none).
563 // Only dimensions [0, d_active) participate in the flat iteration space.
564 // Dimensions [d_active, N) are copied as full extent into the returned box.
565 size_t dim_sizes[N];
566 size_t d_active = N;
567 for (size_t d = 0; d < N; ++d) {
568 dim_sizes[d] = _dims[d].size();
569 if (dim_sizes[d] == 0) { d_active = d; break; }
570 }
571
572 if (d_active == 0) return { *this, 0 }; // outermost dim is zero — no work
573
574 // Decode flat_beg into coords for active dimensions [0, d_active) only.
575 size_t coords[N] = {};
576 size_t temp = flat_beg;
577 for (size_t d = d_active; d-- > 0; ) {
578 coords[d] = temp % dim_sizes[d];
579 temp /= dim_sizes[d];
580 }
581
582 // Find grow_dim within [0, d_active): record each dim BEFORE the budget check
583 // (ceil: round up to the next boundary). Stop when budget met or trailing-zeros fires.
584 size_t grow_dim = d_active - 1;
585 size_t inner_volume = 1;
586 size_t active_inner_vol = 1;
587
588 for (size_t d = d_active; d-- > 0; ) {
589 if (d + 1 < d_active && coords[d + 1] != 0) break; // trailing-zeros rule
590
591 grow_dim = d;
592 active_inner_vol = inner_volume;
593
594 if (inner_volume >= chunk_size) break;
595
596 inner_volume *= dim_sizes[d];
597 }
598
599 // Steps along grow_dim: ceil division, at least 1.
600 size_t steps_left = dim_sizes[grow_dim] - coords[grow_dim];
601 size_t steps_needed = (std::max)(size_t{1}, chunk_size / active_inner_vol);
602 size_t steps_to_take = (std::min)(steps_left, steps_needed);
603
604 // Construct the sub-box. Active dims [0, d_active) are locked/grow/inner
605 // as usual; dims [d_active, N) are copied as full extent (zero-size and beyond).
606 std::array<IndexRange<T, 1>, N> box_dims;
607 for (size_t d = 0; d < N; ++d) {
608 const auto& dim = _dims[d];
609 const T step = dim.step_size();
610 const T beg = dim.begin();
611 if (d >= d_active) {
612 box_dims[d] = dim; // full extent
613 } else if (d < grow_dim) {
614 T b = beg + static_cast<T>(coords[d]) * step;
615 box_dims[d] = IndexRange<T, 1>(b, b + step, step); // locked
616 } else if (d == grow_dim) {
617 box_dims[d] = IndexRange<T, 1>(
618 beg + static_cast<T>(coords[d]) * step,
619 beg + static_cast<T>(coords[d] + steps_to_take) * step,
620 step
621 );
622 } else {
623 box_dims[d] = dim; // full inner extent
624 }
625 }
626
627 return { IndexRange<T, N>(box_dims), steps_to_take * active_inner_vol };
628}

630template <std::integral T, size_t N>
631std::pair<IndexRange<T, N>, size_t>
632IndexRange<T, N>::slice_floor(size_t flat_beg, size_t chunk_size) const {
633
634 if (chunk_size == 0) {
635 return { *this, 0 };
636 }
637
638 // Find d_active: the index of the first zero-size dimension (or N if none).
639 // Only dimensions [0, d_active) participate in the flat iteration space.
640 // Dimensions [d_active, N) are copied as full extent into the returned box.
641 size_t dim_sizes[N];
642 size_t d_active = N;
643 for (size_t d = 0; d < N; ++d) {
644 dim_sizes[d] = _dims[d].size();
645 if (dim_sizes[d] == 0) { d_active = d; break; }
646 }
647
648 if (d_active == 0) return { *this, 0 }; // outermost dim is zero — no work
649
650 // Decode flat_beg into coords for active dimensions [0, d_active) only.
651 size_t coords[N] = {};
652 size_t temp = flat_beg;
653 for (size_t d = d_active; d-- > 0; ) {
654 coords[d] = temp % dim_sizes[d];
655 temp /= dim_sizes[d];
656 }
657
658 // Find grow_dim within [0, d_active): commit each dim AFTER the budget check
659 // (floor: round down to the previous boundary).
660 size_t grow_dim = d_active - 1;
661 size_t active_inner_vol = 1;
662 size_t inner_volume = 1;
663
664 for (size_t d = d_active; d-- > 0; ) {
665 if (d + 1 < d_active && coords[d + 1] != 0) break; // trailing-zeros rule
666
667 size_t next_vol = inner_volume * dim_sizes[d];
668 if (next_vol > chunk_size && inner_volume > 1) break;
669
670 grow_dim = d;
671 active_inner_vol = inner_volume;
672 inner_volume = next_vol;
673 }
674
675 // Steps along grow_dim: floor division, at least 1 for forward progress.
676 size_t steps_left = dim_sizes[grow_dim] - coords[grow_dim];
677 size_t steps_needed = (std::max)(size_t{1}, chunk_size / active_inner_vol);
678 size_t steps_to_take = (std::min)(steps_left, steps_needed);
679
680 // Construct the sub-box. Active dims [0, d_active) are locked/grow/inner
681 // as usual; dims [d_active, N) are copied as full extent (zero-size and beyond).
682 std::array<IndexRange<T, 1>, N> box_dims;
683 for (size_t d = 0; d < N; ++d) {
684 const auto& dim = _dims[d];
685 const T step = dim.step_size();
686 const T beg = dim.begin();
687 if (d >= d_active) {
688 box_dims[d] = dim; // full extent
689 } else if (d < grow_dim) {
690 T b = beg + static_cast<T>(coords[d]) * step;
691 box_dims[d] = IndexRange<T, 1>(b, b + step, step); // locked
692 } else if (d == grow_dim) {
693 box_dims[d] = IndexRange<T, 1>(
694 beg + static_cast<T>(coords[d]) * step,
695 beg + static_cast<T>(coords[d] + steps_to_take) * step,
696 step
697 );
698 } else {
699 box_dims[d] = dim; // full inner extent
700 }
701 }
702
703 return { IndexRange<T, N>(box_dims), steps_to_take * active_inner_vol };
704}
