70template <std::integral T>
71constexpr size_t distance(T beg, T end, T step) {
72 if constexpr (std::is_unsigned_v<T>) {
73 // step is always positive for unsigned types — standard ceiling division
74 return static_cast<size_t>((end - beg + step - T{1}) / step);
75 } else {
76 // signed: step may be positive or negative
77 return static_cast<size_t>((end - beg + step + (step > T{0} ? T{-1} : T{1})) / step);
78 }
79}

320template <std::integral T, size_t N>
321std::array<T, N> IndexRange<T, N>::coords(size_t flat) const {
322 std::array<T, N> c;
323 for (size_t d = N; d-- > 0; ) {
324 size_t sz = _dims[d].size();
325 size_t pos = flat % sz;
326 c[d] = _dims[d].begin() + static_cast<T>(pos) * _dims[d].step_size();
327 flat /= sz;
328 }
329 return c;

332template <std::integral T, size_t N>
333std::pair<IndexRange<T, N>, size_t>
334IndexRange<T, N>::consume_chunk(size_t flat_beg, size_t requested_size) const {
335
336 if (requested_size == 0) {
337 return { *this, 0 };
338 }
339
340 std::array<size_t, N> coords;
341 size_t temp = flat_beg;
342
343 // 1. Convert flat_beg to ND coordinates.
344 // We use the standard unsigned reverse iteration idiom (d-- > 0) to avoid
345 // -Wsign-compare warnings while correctly wrapping around after 0.
346 for (size_t d = N; d-- > 0; ) {
347 size_t dim_size = _dims[d].size();
348 coords[d] = temp % dim_size;
349 temp /= dim_size;
350 }
351
352 // 2. Find the outermost dimension whose cumulative inner volume first meets
353 // or exceeds requested_size (the "minimum overshooting box" rule).
354 //
355 // We walk from the innermost dimension outward, recording grow_dim on
356 // every iteration BEFORE checking the budget. This means grow_dim always
357 // refers to the current candidate and active_inner_vol is its cumulative
358 // volume. We stop as soon as inner_volume >= requested_size (budget met)
359 // or the trailing-zeros rule fires (geometry prevents further growth).
360 //
361 // Trailing-zeros rule: dimension d cannot be the grow dimension if any
362 // dimension inner to it is not at coordinate 0. Growing d in that case
363 // would produce a discontiguous "stair-step" shape rather than an
364 // orthogonal hyper-rectangle. When this fires we break immediately,
365 // returning the best box reachable regardless of requested_size.
366 size_t grow_dim = N - 1;
367 size_t inner_volume = 1;
368 size_t active_inner_vol = 1;
369
370 for (size_t d = N; d-- > 0; ) {
371 // Trailing-zeros rule: an inner dimension is not at coordinate 0.
372 // No outer dimension can grow; return the best box we can form here.
373 if (d + 1 < N && coords[d + 1] != 0) {
374 break;
375 }
376
377 // Record this dimension as the current candidate before the budget check.
378 grow_dim = d;
379 active_inner_vol = inner_volume;
380
381 // Stop as soon as the cumulative volume meets the requested size.
382 // This is the outermost dimension we need — going further would produce
383 // a larger box than necessary.
384 if (inner_volume >= requested_size) {
385 break;
386 }
387
388 // Branchless accumulation: multiply unconditionally. The potential
389 // dead-store when d == 0 is harmless because the loop ends immediately.
390 inner_volume *= _dims[d].size();
391 }
392
393 // 3. Determine how many steps to take along grow_dim.
394 //
395 // active_inner_vol is the volume of one step along grow_dim. When
396 // active_inner_vol >= requested_size (the budget was already met at this
397 // dimension or the trailing-zeros rule fired with active_inner_vol == 1),
398 // integer division yields 0 and std::max promotes it to 1, taking exactly
399 // one indivisible step — the minimum overshooting box.
400 size_t steps_left = _dims[grow_dim].size() - coords[grow_dim];
401 size_t steps_needed = (std::max)(size_t{1}, requested_size / active_inner_vol);
402 size_t steps_to_take = (std::min)(steps_left, steps_needed);
403
404 // 4. Construct the orthogonal sub-box.
405 std::array<IndexRange<T, 1>, N> box_dims;
406 for (size_t d = 0; d < N; ++d) {
407 if (d < grow_dim) {
408 // Outer dimensions are locked at their current coordinate.
409 // We use coords[d] + 1 for the end bound because C++ uses half-open
410 // intervals [begin, end). To lock a dimension to exactly ONE element,
411 // the end bound must be exactly 1 step past the begin bound.
412 box_dims[d] = IndexRange<T, 1>(
413 _dims[d].begin() + static_cast<T>(coords[d]) * _dims[d].step_size(),
414 _dims[d].begin() + static_cast<T>(coords[d] + 1) * _dims[d].step_size(),
415 _dims[d].step_size()
416 );
417 } else if (d == grow_dim) {
418 // The chosen dimension expands by the calculated steps.
419 box_dims[d] = IndexRange<T, 1>(
420 _dims[d].begin() + static_cast<T>(coords[d]) * _dims[d].step_size(),
421 _dims[d].begin() + static_cast<T>(coords[d] + steps_to_take) * _dims[d].step_size(),
422 _dims[d].step_size()
423 );
424 } else {
425 // Inner dimensions take their full extent.
426 box_dims[d] = _dims[d];
427 }
428 }
429
430 return { IndexRange<T, N>(box_dims), steps_to_take * active_inner_vol };
431}
