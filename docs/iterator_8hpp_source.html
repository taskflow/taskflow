70template <std::integral T>
71constexpr size_t distance(T beg, T end, T step) {
72 if constexpr (std::is_unsigned_v<T>) {
73 // step is always positive for unsigned types — standard ceiling division
74 return static_cast<size_t>((end - beg + step - T{1}) / step);
75 } else {
76 // signed: step may be positive or negative
77 return static_cast<size_t>((end - beg + step + (step > T{0} ? T{-1} : T{1})) / step);
78 }
79}

456template <std::integral T, size_t N>
457std::array<T, N> IndexRange<T, N>::coords(size_t flat) const {
458 std::array<T, N> c;
459 for (size_t d = N; d-- > 0; ) {
460 size_t sz = _dims[d].size();
461 size_t pos = flat % sz;
462 c[d] = _dims[d].begin() + static_cast<T>(pos) * _dims[d].step_size();
463 flat /= sz;
464 }
465 return c;

468template <std::integral T, size_t N>
469size_t IndexRange<T, N>::ceil(size_t chunk_size) const {
470 // The natural boundary sizes are the suffix products walked from innermost
471 // outward: 1, dim[N-1], dim[N-1]*dim[N-2], ..., size().
472 // Return the first suffix product that meets or exceeds chunk_size.
473 // If any dimension is zero, size()==0 and we return 0 (no work).
474 if (size() == 0) return 0;
475 size_t inner_volume = 1;
476 if (inner_volume >= chunk_size) return inner_volume;
477 for (size_t d = N; d-- > 0; ) {
478 inner_volume *= _dims[d].size();
479 if (inner_volume >= chunk_size) {
480 return inner_volume;
481 }
482 }
483 return inner_volume; // == size(), the largest possible boundary
484}

486template <std::integral T, size_t N>
487size_t IndexRange<T, N>::floor(size_t chunk_size) const {
488 // Walk suffix products outward; commit each one as long as it fits within
489 // chunk_size. The last committed value is the answer.
490 // If any dimension is zero, size()==0 and we return 0 (no work).
491 if (size() == 0) return 0;
492 size_t inner_volume = 1;
493 size_t last_fit = 1;
494 for (size_t d = N; d-- > 0; ) {
495 size_t next = inner_volume * _dims[d].size();
496 if (next > chunk_size) {
497 break;
498 }
499 inner_volume = next;
500 last_fit = inner_volume;
501 }
502 return last_fit;
503}

505template <std::integral T, size_t N>
506std::pair<IndexRange<T, N>, size_t>
507IndexRange<T, N>::slice_ceil(size_t flat_beg, size_t chunk_size) const {
508
509 if (chunk_size == 0) {
510 return { *this, 0 };
511 }
512
513 // Fused pass: cache dim sizes and decode flat_beg to ND coords in one
514 // backward traversal. Also detects a zero-size dimension early (empty range).
515 // A single backward pass suffices because the coord decode is naturally
516 // innermost-first (temp %= dim_size[d] then /= dim_size[d]).
517 size_t dim_sizes[N];
518 size_t coords[N];
519 size_t temp = flat_beg;
520 for (size_t d = N; d-- > 0; ) {
521 dim_sizes[d] = _dims[d].size();
522 if (dim_sizes[d] == 0) return { *this, 0 }; // empty range
523 coords[d] = temp % dim_sizes[d];
524 temp /= dim_sizes[d];
525 }
526
527 // Find grow_dim: walk innermost→outermost, record each dim as candidate
528 // BEFORE the budget check (ceil: round up to the next boundary).
529 // Stop when inner_volume meets chunk_size or trailing-zeros fires.
530 size_t grow_dim = N - 1;
531 size_t inner_volume = 1;
532 size_t active_inner_vol = 1;
533
534 for (size_t d = N; d-- > 0; ) {
535 if (d + 1 < N && coords[d + 1] != 0) break; // trailing-zeros rule
536
537 grow_dim = d;
538 active_inner_vol = inner_volume;
539
540 if (inner_volume >= chunk_size) break;
541
542 inner_volume *= dim_sizes[d];
543 }
544
545 // Steps along grow_dim: ceil division, at least 1.
546 size_t steps_left = dim_sizes[grow_dim] - coords[grow_dim];
547 size_t steps_needed = (std::max)(size_t{1}, chunk_size / active_inner_vol);
548 size_t steps_to_take = (std::min)(steps_left, steps_needed);
549
550 // Construct the orthogonal sub-box.
551 std::array<IndexRange<T, 1>, N> box_dims;
552 for (size_t d = 0; d < N; ++d) {
553 const auto& dim = _dims[d];
554 const T step = dim.step_size();
555 const T beg = dim.begin();
556 if (d < grow_dim) {
557 T b = beg + static_cast<T>(coords[d]) * step;
558 box_dims[d] = IndexRange<T, 1>(b, b + step, step);
559 } else if (d == grow_dim) {
560 box_dims[d] = IndexRange<T, 1>(
561 beg + static_cast<T>(coords[d]) * step,
562 beg + static_cast<T>(coords[d] + steps_to_take) * step,
563 step
564 );
565 } else {
566 box_dims[d] = dim;
567 }
568 }
569
570 return { IndexRange<T, N>(box_dims), steps_to_take * active_inner_vol };
571}

573template <std::integral T, size_t N>
574std::pair<IndexRange<T, N>, size_t>
575IndexRange<T, N>::slice_floor(size_t flat_beg, size_t chunk_size) const {
576
577 if (chunk_size == 0) {
578 return { *this, 0 };
579 }
580
581 // Fused pass: cache dim sizes and decode flat_beg to ND coords in one
582 // backward traversal. Also detects a zero-size dimension early (empty range).
583 size_t dim_sizes[N];
584 size_t coords[N];
585 size_t temp = flat_beg;
586 for (size_t d = N; d-- > 0; ) {
587 dim_sizes[d] = _dims[d].size();
588 if (dim_sizes[d] == 0) return { *this, 0 }; // empty range
589 coords[d] = temp % dim_sizes[d];
590 temp /= dim_sizes[d];
591 }
592
593 // Find grow_dim: walk innermost→outermost, commit each dim AFTER checking
594 // the budget (floor: round down to the previous boundary).
595 // Stop when next_vol would exceed chunk_size (and inner > 1 to ensure
596 // the innermost dim is always committed as the minimum fallback).
597 size_t grow_dim = N - 1;
598 size_t active_inner_vol = 1;
599 size_t inner_volume = 1;
600
601 for (size_t d = N; d-- > 0; ) {
602 if (d + 1 < N && coords[d + 1] != 0) break; // trailing-zeros rule
603
604 size_t next_vol = inner_volume * dim_sizes[d];
605 if (next_vol > chunk_size && inner_volume > 1) break;
606
607 grow_dim = d;
608 active_inner_vol = inner_volume;
609 inner_volume = next_vol;
610 }
611
612 // Steps along grow_dim: floor division, at least 1 for forward progress.
613 size_t steps_left = dim_sizes[grow_dim] - coords[grow_dim];
614 size_t steps_needed = (std::max)(size_t{1}, chunk_size / active_inner_vol);
615 size_t steps_to_take = (std::min)(steps_left, steps_needed);
616
617 // Construct the orthogonal sub-box.
618 std::array<IndexRange<T, 1>, N> box_dims;
619 for (size_t d = 0; d < N; ++d) {
620 const auto& dim = _dims[d];
621 const T step = dim.step_size();
622 const T beg = dim.begin();
623 if (d < grow_dim) {
624 T b = beg + static_cast<T>(coords[d]) * step;
625 box_dims[d] = IndexRange<T, 1>(b, b + step, step);
626 } else if (d == grow_dim) {
627 box_dims[d] = IndexRange<T, 1>(
628 beg + static_cast<T>(coords[d]) * step,
629 beg + static_cast<T>(coords[d] + steps_to_take) * step,
630 step
631 );
632 } else {
633 box_dims[d] = dim;
634 }
635 }
636
637 return { IndexRange<T, N>(box_dims), steps_to_take * active_inner_vol };
638}
