352constexpr size_t coprime(size_t N) {
353 if(N < 3) {
354 return 1;
355 }
356 for (size_t x = N; --x > 0;) {
357 if (std::gcd(x, N) == 1) {
358 return x;
359 }
360 }
361 return 1;
362}
