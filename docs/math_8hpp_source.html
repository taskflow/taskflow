264constexpr size_t coprime(size_t N) {
265 if(N < 3) {
266 return 1;
267 }
268 for (size_t x = N; --x > 0;) {
269 if (std::gcd(x, N) == 1) {
270 return x;
271 }
272 }
273 return 1;
