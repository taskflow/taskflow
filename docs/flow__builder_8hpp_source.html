1444// Function: erase
1445inline void FlowBuilder::erase(Task task) {
1446
1447 if (!task._node) {
1448 return;
1449 }
1450
1451 // remove task from its successors' predecessor list
1452 for(size_t i=0; i<task._node->_num_successors; ++i) {
1453 task._node->_edges[i]->_remove_predecessors(task._node);
1454 }
1455
1456 // remove task from its precedessors' successor list
1457 for(size_t i=task._node->_num_successors; i<task._node->_edges.size(); ++i) {
1458 task._node->_edges[i]->_remove_successors(task._node);
1459 }
1460
1461 _graph._erase(task._node);
1462}

1465// Procedure: _linearize
1466template <typename L>
1467void FlowBuilder::_linearize(L& keys) {
1468
1469 auto itr = keys.begin();
1470 auto end = keys.end();
1471
1472 if(itr == end) {
1473 return;
1474 }
1475
1476 auto nxt = itr;
1477
1478 for(++nxt; nxt != end; ++nxt, ++itr) {
1479 itr->_node->_precede(nxt->_node);
1480 }
1481}
