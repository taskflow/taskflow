1512// Function: erase
1513inline void FlowBuilder::erase(Task task) {
1514
1515 if (!task._node) {
1516 return;
1517 }
1518
1519 // remove task from its successors' predecessor list
1520 for(size_t i=0; i<task._node->_num_successors; ++i) {
1521 task._node->_edges[i]->_remove_predecessors(task._node);
1522 }
1523
1524 // remove task from its precedessors' successor list
1525 for(size_t i=task._node->_num_successors; i<task._node->_edges.size(); ++i) {
1526 task._node->_edges[i]->_remove_successors(task._node);
1527 }
1528
1529 _graph._erase(task._node);
1530}

1533// Procedure: _linearize
1534template <typename L>
1535void FlowBuilder::_linearize(L& keys) {
1536
1537 auto itr = keys.begin();
1538 auto end = keys.end();
1539
1540 if(itr == end) {
1541 return;
1542 }
1543
1544 auto nxt = itr;
1545
1546 for(++nxt; nxt != end; ++nxt, ++itr) {
1547 itr->_node->_precede(nxt->_node);
1548 }
1549}
