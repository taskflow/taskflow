280 cudaTask task, dim3 g, dim3 b, size_t s, F f, ArgsT... args
281) {
282 cudaKernelNodeParams p;

284 void* arguments[sizeof...(ArgsT)] = { (void*)(&args)... };
285 p.func = (void*)f;

287 p.blockDim = b;
288 p.sharedMemBytes = s;
289 p.kernelParams = arguments;
